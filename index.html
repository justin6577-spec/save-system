Quick Start

// Get the save manager (works from anywhere)
USaveManager* SaveMgr = GetGameInstance()->GetSubsystem<USaveManager>();

// Save
SaveMgr->SaveToSlot("AutoSave", MySaveObject, "Level_5");

// Load
SaveMgr->LoadFromSlot("AutoSave");

// List all saves
TArray<FSaveSlotInfo> Slots = SaveMgr->GetAllSaveSlots();
